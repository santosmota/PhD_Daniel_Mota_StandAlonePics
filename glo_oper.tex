	{\ensuremath{*}}%symbol %\left(\right)^*

%% D
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\glsxtrnewsymbol
	[description={infinitesimal variation of a variable, i.e., $\partial x$; used in the context of partial derivatives}, %toerhwhen used as $\partial f / \partial x$ denotes the partial derivative of the function $f$ with respect to the scalar $x$}, % If $f$ is a vector, the operator is applied to each of the vector's components},
	sort=del,
	parent=operators]%description
	{del}%label
	{\ensuremath{\partial}}%symbol

\glsxtrnewsymbol
	[description={finite variation of a variable, i.e., $\Delta x$}, % indicates a small finite variation of the variable $x$}, % If $f$ is a vector, the operator is applied to each of the vector's components},
	sort=delta,
	parent=operators]%description
	{deltavariation}%label
	{\ensuremath{\Delta}}%symbol
	
\glsxtrnewsymbol
	[description={infinitesimal variation of a variable, i.e., $\inftsml x$}, %; when used as $\frac{\mathrm{d}f}{\mathrm{d}x}$ denotes the derivative of the function $f$ with respect to the variable $x$}, % art of a variable as in $\mathrm{d}x$     If $x$ or $f$ are a vector, the operator is applied to each of the vector's components
	sort=d,
	parent=operators]%description
	{dinfinitesimal}%label
	{\ensuremath{\mathrm{d}}}%symbol

\glsxtrnewsymbol
	[description={dot product of two column vectors $\matr{x}$ and $\matr{y}$ is $\matr{x} \cdot \matr{y} = \matr{x}^{\gls{transpose}} \matr{y}$.},
	sort=DotProduct,
	parent=operators]%description
	{dotproduct}%label
	{\ensuremath{\cdot}}%symbol   % \bullet
